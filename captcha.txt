
<script setup>

import { usePrenotazioni } from '~/store/prenotazioni';
import  { useAuth } from '../../store/auth';
const prenotazioniStore = usePrenotazioni();


  import { ref } from 'vue';
  import Captcha from '../components/Captcha/Captcha.vue';
  
  const userInput = ref('');
  const captchaRef = ref(null);
  const result = ref(null);
  
  function checkCaptcha() {
    result.value = captchaRef.value.verifyCaptcha(userInput.value);
  }
</script>


<template>
  
  <div style = "margin-left: 500px;">
      <h2>Verifica Captcha</h2>
      <Captcha ref="captchaRef" />
      <input v-model="userInput" placeholder="Inserisci il captcha" />
      <button @click="checkCaptcha">Verifica</button>
      <p v-if="result !== null">{{ result ? '✅ Corretto!' : '❌ Errato!' }}</p>
  </div>
</template>
